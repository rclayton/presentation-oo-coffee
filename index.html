Functional
uses evaluation of mathematical functions
and avoids state and mutable data

Behavior is described as functions:

function square(x){ return x * x; }

State passed into the function remains unchanged:

var x = 5;
var y = square(x);
console.log(x); // 5
console.log(y); // 25

Object-Oriented
uses data structures consisting of
data fields and methods together

Behavior is defined on the container:

var Car = function(){

 this.accelerate = function(mph){
  speed += mph;

Behavior of the container modifies state:

var car = new Car();
car.accelerate(10); // 10mph
car.accelerate(15); // 25mph

Object-Oriented JavaScript
Confusing, underused, and underappreciated.

What's Confusing about OO.js?

  • No class declaration syntax.
  • Unfamiliar inheritance model.
  • Limited encapsulation.

No class declaration syntax

Class/Type declaration is simply assigning properties and functions on another function (which is also strangely an object)!

var Car = function(gasAmount){
 // This is the constructor!
 this.speed = 0; // Public property
 this.gas = (gasAmount > 10.5)? 10.5 : gasAmount;
 // Public method
 this.fillTank = function(gasAmount){
  if ((this.gas + gasAmount) >= 10.5){ this.gas = 10.5; }
  else { this.gas += gasAmount; }
 }
};
//Public method
Car.prototype.move = function(){
 if (this.gas > 0){ this.gas -= 0.1; this.speed++; }
};

Unfamiliar inheritance model.

Prototypical Inheritance

  • Objects inherit directly from other objects.
  • Members are "resolved" through a prototype chain.
var Car = function(){
 console.log("Vroom!");
 this.speed = 0;
};

Car.prototype.move = function(){ console.log("Moving"); };

var Honda = function(){
 Car.call(this); // Call the Parent Contructor
 console.log("Gas efficient, Vroom!");
};

Honda.prototype = Object.create(Car.prototype);

var civic = new Honda();
civic.move(); // Resolved to Car

Limited encapsulation

Member visibility is a real problem in JavaScript, since there is no real concept of private or protected state.

var Clazz = function(){
 // Crockford calls these 'private', I call them 'hidden'.
 var hidden = true;
 // Can access variables in the constructor.
 // Crockford calls this a "privileged method"
 var modifyInternal = function(value){ hidden = value; }
 // Public variable
 this.publicVar = 10;
 // 'Public-privileged' method since it has access to
 // variables defined within the constructor.
 this.publicMethod = function(){}
}
// Public method attached to the prototype.
Clazz.prototype.move = function(){
 // Has no access to variables defined within the constructor
};


Douglas Crockford would disagree with me to some extent. You can see a more indepth article about member visibility here: http://www.crockford.com/javascript/private.html

Object-Oriented CoffeeScript

Instead of repetitively attaching functions to a prototype, CoffeeScript provides a basic class structure that allows you to name your class, set the superclass, assign prototypal properties, and define the constructor, in a single assignable expression.

Inheritance, cont.

var Human, Knight, knight1, knight2, serf,
__hasProp = {}.hasOwnProperty,
__extends = function(child, parent) {
 for (var key in parent) {
  if (__hasProp.call(parent, key)) child[key] = parent[key];
 }
 function ctor() { this.constructor = child; }
 ctor.prototype = parent.prototype;
 child.prototype = new ctor();
 child.__super__ = parent.prototype; return child;
};

Inheritance, cont.

Human = (function() {

  hitPoints = 10;

 return Human;

Inheritance, cont.

Knight = (function(_super) {
 __extends(Knight, _super);
 function Knight() {

 Knight.prototype.attack = function(human) {

 return Knight;

Function Binding

Losing closure-scope sucks and is especially
problematic when using classes:

class TextDestroyerOfDoom
 message: "Your text was destroyed!"
 destroyText: (i, element) ->
  $(element).text @message

destroyer = new TextDestroyerOfDoom()

$("div").each destroyer.destroyText

This code does not work!

this has been overridden by jQuery to the
current DOM element in the iteration,
meaning @message is undefined.

Fat Arrow =>

Binds the execution of the member method to the object instance, preserving the scope of this.

class TextDestroyerOfDoom
message: "Your text was destroyed!"
destroyText: (i, element) =>
	$(element).text @message

destroyer = new TextDestroyerOfDoom()

$("div").each destroyer.destroyText

Fat Arrow, Behind the Scenes

var TextDestroyerOfDoom, destroyer,
 __bind = function(fn, me){
  return function(){ return fn.apply(me, arguments); }; };

TextDestroyerOfDoom = (function() {
 function TextDestroyerOfDoom() {
  this.destroyText = __bind(this.destroyText, this);
 }

 TextDestroyerOfDoom.prototype.message = "Your text was destroyed!";

 TextDestroyerOfDoom.prototype.destroyText = function(i, element) {
  return $(element).text(this.message);
 };

 return TextDestroyerOfDoom;
})();
